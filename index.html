google_sheets_script.js 
function onEdit(e) {
  var sheet = e.range.getSheet();
  
  // Check if the edited cell is D21 (thisWeek)
  if (e.range.getA1Notation() === "D21") {
    var thisWeekCell = sheet.getRange("D21");
    var lastWeekCell = sheet.getRange("E21");
    var totalCell = sheet.getRange("G21");

    var thisWeek = thisWeekCell.getValue();
    var currentTotal = totalCell.getValue();

    if (currentTotal === "") {
      totalCell.setValue(thisWeek);
      lastWeekCell.setValue(0);
    } else {
      lastWeekCell.setValue(currentTotal);
      totalCell.setValue(currentTotal + thisWeek);
    }
  }
}
